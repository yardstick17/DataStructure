[docs]def kadane(list_of_number):
    max_sum = 0
    temporal_sum = 0
    for element in list_of_number:
        temporal_sum += element

        if temporal_sum < 0:
            temporal_sum = 0

        elif max_sum < temporal_sum:
            max_sum = temporal_sum

    return max_sum
